#!/bin/bash

# Quick Start Guide for Portfolio - Windows Users

# If you're on Windows, you can ignore this file and use these commands in PowerShell or Command Prompt

# Step 1: Install Dependencies

npm install

# Step 2: Start Development Server

npm run dev

# Step 3: Open browser at http://localhost:5173

# Step 4: Edit src/data/portfolio.js with your information

# Step 5: Build for production

npm run build

# Step 6: Deploy

# Choose one:

# - GitHub Pages: npm run deploy

# - Vercel: Connect repository to vercel.com

# - Netlify: Connect repository to netlify.com

# - Firebase: firebase deploy
